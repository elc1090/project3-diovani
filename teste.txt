function salvarAnotacao(videoId) {
    var anotacaoTextarea = document.querySelector(".video-container[data-video-id='" + videoId + "'] textarea");
    var anotacaoTexto = anotacaoTextarea.value;

    if (anotacaoTexto.trim() === "") {
        alert("Por favor, insira uma anotação válida.");
        return;
    }

    var video = videos.find(function(item) {
        return item.id === videoId;
    });

    if (video) {
        video.anotacoes.push(anotacaoTexto);
        exibirAnotacoes(videoId);
    }

    anotacaoTextarea.value = "";
}



function exibirAnotacoes(videoId) {
    var anotacoesDiv = document.querySelector(".video-container[data-video-id='" + videoId + "'] .anotacoes");

    if (anotacoesDiv) {
        anotacoesDiv.innerHTML = "";

        var video = videos.find(function(item) {
            return item.id === videoId;
        });

        if (video) {
            video.anotacoes.forEach(function(anotacao) {
                var p = document.createElement("p");
                p.textContent = anotacao;
                anotacoesDiv.appendChild(p);
            });
        }
    }
}
